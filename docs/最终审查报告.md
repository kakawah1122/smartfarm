// 方案1: 替换为真实电话号码（推荐）
contactVet() {
  wx.showModal({
    title: '联系专业兽医',
    content: '是否拨打兽医服务热线：400-123-4567？', // 替换为真实号码
    confirmText: '拨打',
    success: (res) => {
      if (res.confirm) {
        wx.makePhoneCall({
          phoneNumber: '400-123-4567', // 替换为真实号码
          fail: () => {
            wx.showToast({
              title: '拨打电话失败',
              icon: 'none'
            })
          }
        })
      }
    }
  })
}

// 方案2: 从配置中读取（更灵活）
contactVet() {
  const app = getApp()
  const vetPhone = app.globalData.vetPhoneNumber || '400-123-4567'
  
  wx.showModal({
    title: '联系专业兽医',
    content: `是否拨打兽医服务热线：${vetPhone}？`,
    confirmText: '拨打',
    success: (res) => {
      if (res.confirm) {
        wx.makePhoneCall({
          phoneNumber: vetPhone,
          fail: () => {
            wx.showToast({
              title: '拨打电话失败',
              icon: 'none'
            })
          }
        })
      }
    }
  })
}
